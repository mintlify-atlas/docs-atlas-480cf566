## Technical Implementation

### Sales Forecasting with Prophet

<CodeGroup>

```python Data Preparation
import pandas as pd
import numpy as np
from fbprophet import Prophet
import matplotlib.pyplot as plt

# Load and prepare sales data
df = pd.read_csv('sales_data.csv')
df['date'] = pd.to_datetime(df['date'])

# Prepare data for Prophet (requires 'ds' and 'y' columns)
df_prophet = df[['date', 'sales']].rename(
    columns={'date': 'ds', 'sales': 'y'}
)

# Handle outliers and missing values
df_prophet = df_prophet.dropna()
df_prophet['y'] = np.where(
    np.abs(df_prophet['y'] - df_prophet['y'].mean()) > 3 * df_prophet['y'].std(),
    df_prophet['y'].median(),
    df_prophet['y']
)

print(f"Dataset shape: {df_prophet.shape}")
print(f"Date range: {df_prophet['ds'].min()} to {df_prophet['ds'].max()}")
```

```python Model Training
# Initialize and configure Prophet model
model = Prophet(
    yearly_seasonality=True,
    weekly_seasonality=True,
    daily_seasonality=False,
    seasonality_mode='multiplicative',
    changepoint_prior_scale=0.05
)

# Add custom seasonalities
model.add_seasonality(
    name='monthly',
    period=30.5,
    fourier_order=5
)

# Add holiday effects
holidays = pd.DataFrame({
    'holiday': 'special_events',
    'ds': pd.to_datetime(['2024-12-25', '2025-01-01', '2025-07-09']),
    'lower_window': -1,
    'upper_window': 1,
})
model.add_country_holidays(country_name='AR')

# Train the model
print("Training model...")
model.fit(df_prophet)
print("Model training complete!")
```

```python Forecasting & Analysis
# Generate future dates for prediction
future = model.make_future_dataframe(periods=90, freq='D')

# Generate forecast
forecast = model.predict(future)

# Extract key predictions
predictions = forecast[['ds', 'yhat', 'yhat_lower', 'yhat_upper']].tail(90)

# Calculate confidence intervals
predictions['confidence_range'] = (
    predictions['yhat_upper'] - predictions['yhat_lower']
)

# Visualize results
fig1 = model.plot(forecast)
plt.title('Sales Forecast - Next 90 Days')
plt.xlabel('Date')
plt.ylabel('Sales')
plt.tight_layout()
plt.savefig('forecast_plot.png', dpi=300)

# Plot components
fig2 = model.plot_components(forecast)
plt.tight_layout()
plt.savefig('components_plot.png', dpi=300)

print("\nForecast Summary:")
print(f"Average predicted sales: ${predictions['yhat'].mean():,.2f}")
print(f"Peak sales date: {predictions.loc[predictions['yhat'].idxmax(), 'ds']}")
print(f"Peak sales value: ${predictions['yhat'].max():,.2f}")
```

```python Model Evaluation
from sklearn.metrics import mean_absolute_error, mean_squared_error
import numpy as np

# Cross-validation
from fbprophet.diagnostics import cross_validation, performance_metrics

df_cv = cross_validation(
    model,
    initial='730 days',
    period='180 days',
    horizon='90 days'
)

# Calculate performance metrics
df_performance = performance_metrics(df_cv)

print("\nModel Performance Metrics:")
print(f"MAE: ${df_performance['mae'].mean():,.2f}")
print(f"RMSE: ${df_performance['rmse'].mean():,.2f}")
print(f"MAPE: {df_performance['mape'].mean():.2%}")

# Calculate accuracy
accuracy = 1 - df_performance['mape'].mean()
print(f"\nModel Accuracy: {accuracy:.2%}")

# Save model and results
import pickle
with open('sales_forecast_model.pkl', 'wb') as f:
    pickle.dump(model, f)

forecast.to_csv('sales_forecast_output.csv', index=False)
print("\nModel and forecast saved successfully!")
```

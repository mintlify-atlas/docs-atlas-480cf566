## Git Version Control

### Core Git Workflow

Efficient version control for collaborative development:

<Tabs>
  <Tab title="Basic Commands">
    ```bash
    # Initialize repository
    git init
    git remote add origin https://github.com/username/repo.git

    # Daily workflow
    git status                    # Check current status
    git add .                     # Stage all changes
    git add file.js              # Stage specific file
    git commit -m "Add feature"  # Commit with message
    git push origin main         # Push to remote

    # Pull latest changes
    git pull origin main         # Fetch and merge
    git fetch origin             # Fetch without merge
    git merge origin/main        # Merge fetched changes

    # View history
    git log                      # Full commit history
    git log --oneline            # Condensed history
    git log --graph              # Visual branch graph
    git show <commit-hash>       # Show specific commit
    ```
  </Tab>
  <Tab title="Branching">
    ```bash
    # Branch management
    git branch                          # List branches
    git branch feature/new-feature      # Create branch
    git checkout feature/new-feature    # Switch branch
    git checkout -b feature/login       # Create and switch

    # Modern git switch (Git 2.23+)
    git switch main                     # Switch to main
    git switch -c feature/dashboard     # Create and switch

    # Merge branches
    git checkout main
    git merge feature/new-feature       # Merge into main
    git branch -d feature/new-feature   # Delete merged branch

    # Rebase workflow
    git checkout feature/my-feature
    git rebase main                     # Rebase on main
    git push --force-with-lease         # Safe force push

    # View branch differences
    git diff main..feature/new-feature  # Compare branches
    ```
  </Tab>
  <Tab title="Advanced">
    ```bash
    # Stash changes
    git stash                           # Save changes temporarily
    git stash list                      # View stashed changes
    git stash pop                       # Apply and remove stash
    git stash apply                     # Apply keeping stash
    git stash drop                      # Delete stash

    # Undo changes
    git reset HEAD~1                    # Undo last commit (keep changes)
    git reset --hard HEAD~1             # Undo last commit (discard changes)
    git revert <commit-hash>            # Create reverting commit
    git checkout -- file.js             # Discard file changes

    # Interactive rebase
    git rebase -i HEAD~3                # Edit last 3 commits
    # Options: pick, reword, edit, squash, fixup, drop

    # Cherry-pick commits
    git cherry-pick <commit-hash>       # Apply specific commit

    # Clean untracked files
    git clean -n                        # Preview files to delete
    git clean -fd                       # Delete untracked files/dirs

    # Search commits
    git log --grep="bug fix"            # Search commit messages
    git log -S"function name"           # Search code changes
    ```
  </Tab>
  <Tab title="Configuration">
    ```bash
    # User configuration
    git config --global user.name "George Valle"
    git config --global user.email "georgevalle@outlook.com.ar"

    # Editor setup
    git config --global core.editor "code --wait"

    # Aliases for efficiency
    git config --global alias.st status
    git config --global alias.co checkout
    git config --global alias.br branch
    git config --global alias.cm "commit -m"
    git config --global alias.lg "log --graph --oneline --decorate"

    # Default branch name
    git config --global init.defaultBranch main

    # Line ending handling
    git config --global core.autocrlf input  # Mac/Linux
    git config --global core.autocrlf true   # Windows

    # View all config
    git config --list
    git config --global --list
    ```
  </Tab>
</Tabs>

## Vercel Deployment

### Modern Frontend Deployment

Seamless deployment for React applications and serverless functions:

<Tabs>
  <Tab title="Setup & Deploy">
    ```bash
    # Install Vercel CLI
    npm install -g vercel

    # Login to Vercel
    vercel login

    # Deploy project
    vercel                      # Deploy to preview
    vercel --prod               # Deploy to production

    # Link existing project
    vercel link

    # Environment variables
    vercel env add              # Add environment variable
    vercel env ls               # List variables
    vercel env rm               # Remove variable

    # View logs
    vercel logs                 # Recent logs
    vercel logs [deployment-url]  # Specific deployment

    # List deployments
    vercel ls

    # Remove deployment
    vercel rm [deployment-id]
    ```
  </Tab>
  <Tab title="Configuration">
    ```json
    // vercel.json
    {
      "version": 2,
      "builds": [
        {
          "src": "package.json",
          "use": "@vercel/next"
        }
      ],
      "routes": [
        {
          "src": "/api/(.*)",
          "dest": "/api/$1"
        },
        {
          "src": "/(.*)",
          "dest": "/$1"
        }
      ],
      "env": {
        "MONGODB_URI": "@mongodb-uri",
        "API_KEY": "@api-key"
      },
      "headers": [
        {
          "source": "/api/(.*)",
          "headers": [
            {
              "key": "Access-Control-Allow-Origin",
              "value": "*"
            },
            {
              "key": "Access-Control-Allow-Methods",
              "value": "GET, POST, PUT, DELETE, OPTIONS"
            }
          ]
        }
      ],
      "redirects": [
        {
          "source": "/old-page",
          "destination": "/new-page",
          "permanent": true
        }
      ]
    }
    ```
  </Tab>
  <Tab title="Serverless Functions">
    ```javascript
    // api/hello.js
    module.exports = (req, res) => {
      const { name = 'World' } = req.query;
      res.status(200).json({ message: `Hello, ${name}!` });
    };

    // api/products.js
    import { MongoClient } from 'mongodb';

    export default async function handler(req, res) {
      const client = await MongoClient.connect(process.env.MONGODB_URI);
      const db = client.db('store');
      
      if (req.method === 'GET') {
        const products = await db.collection('products').find({}).toArray();
        res.status(200).json(products);
      } else if (req.method === 'POST') {
        const product = req.body;
        const result = await db.collection('products').insertOne(product);
        res.status(201).json(result);
      }
      
      await client.close();
    }

    // api/auth/[...auth].js - Catch-all route
    export default function handler(req, res) {
      const { auth } = req.query;  // Array of route segments
      res.json({ path: auth });
    }
    ```
  </Tab>

```python
import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.data import Dataset, DataLoader
import torchaudio
import numpy as np

class VoiceEncoder(nn.Module):
    """Neural network encoder for voice embedding extraction"""
    
    def __init__(self, input_dim=80, hidden_dim=256, embedding_dim=128):
        super(VoiceEncoder, self).__init__()
        
        self.conv_layers = nn.Sequential(
            nn.Conv1d(input_dim, 128, kernel_size=5, padding=2),
            nn.BatchNorm1d(128),
            nn.ReLU(),
            nn.MaxPool1d(2),
            
            nn.Conv1d(128, 256, kernel_size=5, padding=2),
            nn.BatchNorm1d(256),
            nn.ReLU(),
            nn.MaxPool1d(2),
            
            nn.Conv1d(256, 512, kernel_size=5, padding=2),
            nn.BatchNorm1d(512),
            nn.ReLU(),
            nn.MaxPool1d(2),
        )
        
        self.lstm = nn.LSTM(
            input_size=512,
            hidden_size=hidden_dim,
            num_layers=3,
            batch_first=True,
            dropout=0.3,
            bidirectional=True
        )
        
        self.fc = nn.Linear(hidden_dim * 2, embedding_dim)
        
    def forward(self, x):
        # x shape: (batch, features, time)
        x = self.conv_layers(x)
        
        # Reshape for LSTM: (batch, time, features)
        x = x.transpose(1, 2)
        
        # LSTM processing
        lstm_out, (hidden, cell) = self.lstm(x)
        
        # Use final hidden state
        # Concatenate forward and backward directions
        hidden = torch.cat([hidden[-2], hidden[-1]], dim=1)
        
        # Project to embedding space
        embedding = self.fc(hidden)
        
        # L2 normalize
        embedding = F.normalize(embedding, p=2, dim=1)
        
        return embedding

class VoiceDecoder(nn.Module):
    """Neural network decoder for audio generation"""
    
    def __init__(self, embedding_dim=128, hidden_dim=512, output_dim=80):
        super(VoiceDecoder, self).__init__()
        
        self.fc1 = nn.Linear(embedding_dim, hidden_dim)
        
        self.lstm = nn.LSTM(
            input_size=hidden_dim,
            hidden_size=hidden_dim,
            num_layers=3,
            batch_first=True,
            dropout=0.3
        )
        
        self.attention = nn.MultiheadAttention(
            embed_dim=hidden_dim,
            num_heads=8,
            dropout=0.1
        )
        
        self.conv_transpose = nn.Sequential(
            nn.ConvTranspose1d(hidden_dim, 256, kernel_size=4, stride=2, padding=1),
            nn.BatchNorm1d(256),
            nn.ReLU(),
            
            nn.ConvTranspose1d(256, 128, kernel_size=4, stride=2, padding=1),
            nn.BatchNorm1d(128),
            nn.ReLU(),
            
            nn.ConvTranspose1d(128, output_dim, kernel_size=4, stride=2, padding=1),
        )
        
    def forward(self, embedding, target_length):
        # Project embedding and repeat for sequence
        x = self.fc1(embedding)
        x = x.unsqueeze(1).repeat(1, target_length, 1)
        
        # LSTM processing
        lstm_out, _ = self.lstm(x)
        
        # Self-attention
        attn_out, _ = self.attention(lstm_out, lstm_out, lstm_out)
        
        # Transpose for conv: (batch, features, time)
        attn_out = attn_out.transpose(1, 2)
        
        # Generate output
        output = self.conv_transpose(attn_out)
        
        return output

# Initialize models
device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
encoder = VoiceEncoder().to(device)
decoder = VoiceDecoder().to(device)

print(f"Models initialized on {device}")
print(f"Encoder parameters: {sum(p.numel() for p in encoder.parameters()):,}")
print(f"Decoder parameters: {sum(p.numel() for p in decoder.parameters()):,}")
```

    ```python
    import torch
    import torchaudio
    import librosa
    import numpy as np
    from pathlib import Path

    class AudioProcessor:
        """Process audio files for voice cloning"""
        
        def __init__(self, sample_rate=22050, n_mels=80, n_fft=1024, hop_length=256):
            self.sample_rate = sample_rate
            self.n_mels = n_mels
            self.n_fft = n_fft
            self.hop_length = hop_length
            
            self.mel_transform = torchaudio.transforms.MelSpectrogram(
                sample_rate=sample_rate,
                n_fft=n_fft,
                hop_length=hop_length,
                n_mels=n_mels
            )
        
        def load_audio(self, audio_path):
            """Load and preprocess audio file"""
            waveform, sr = torchaudio.load(audio_path)
            
            # Resample if necessary
            if sr != self.sample_rate:
                resampler = torchaudio.transforms.Resample(sr, self.sample_rate)
                waveform = resampler(waveform)
            
            # Convert to mono if stereo
            if waveform.shape[0] > 1:
                waveform = torch.mean(waveform, dim=0, keepdim=True)
            
            return waveform
        
        def extract_mel_spectrogram(self, waveform):
            """Extract mel spectrogram features"""
            mel_spec = self.mel_transform(waveform)
            
            # Convert to log scale
            mel_spec = torch.log(mel_spec + 1e-9)
            
            # Normalize
            mel_spec = (mel_spec - mel_spec.mean()) / (mel_spec.std() + 1e-9)
            
            return mel_spec
        
        def process_audio(self, audio_path):
            """Complete audio processing pipeline"""
            waveform = self.load_audio(audio_path)
            mel_spec = self.extract_mel_spectrogram(waveform)
            return mel_spec
        
        def trim_silence(self, waveform, threshold=0.02):
            """Remove silence from beginning and end"""
            waveform_np = waveform.numpy().squeeze()
            trimmed, _ = librosa.effects.trim(waveform_np, top_db=threshold)
            return torch.from_numpy(trimmed).unsqueeze(0)

    class VoiceDataset(torch.utils.data.Dataset):
        """Dataset for voice cloning training"""
        
        def __init__(self, audio_dir, processor, max_length=256):
            self.audio_files = list(Path(audio_dir).glob('**/*.wav'))
            self.processor = processor
            self.max_length = max_length
        
        def __len__(self):
            return len(self.audio_files)
        
        def __getitem__(self, idx):
            audio_path = self.audio_files[idx]
            
            # Process audio
            mel_spec = self.processor.process_audio(str(audio_path))
            
            # Pad or truncate to fixed length
            if mel_spec.shape[2] < self.max_length:
                pad_length = self.max_length - mel_spec.shape[2]
                mel_spec = F.pad(mel_spec, (0, pad_length))
            else:
                mel_spec = mel_spec[:, :, :self.max_length]
            
            return mel_spec.squeeze(0), str(audio_path.stem)

    # Initialize processor and dataset
    processor = AudioProcessor()
    dataset = VoiceDataset('data/voice_samples', processor)
    dataloader = DataLoader(dataset, batch_size=16, shuffle=True, num_workers=4)

    print(f"Dataset size: {len(dataset)} samples")
    ```

    ```python
    import torch
    import torch.nn as nn
    import torch.optim as optim
    from torch.utils.tensorboard import SummaryWriter
    from tqdm import tqdm

    class VoiceCloningTrainer:
        """Training pipeline for voice cloning model"""
        
        def __init__(self, encoder, decoder, device='cuda'):
            self.encoder = encoder.to(device)
            self.decoder = decoder.to(device)
            self.device = device
            
            # Optimizers
            self.optimizer_encoder = optim.AdamW(
                self.encoder.parameters(),
                lr=1e-4,
                betas=(0.9, 0.999),
                weight_decay=1e-5
            )
            
            self.optimizer_decoder = optim.AdamW(
                self.decoder.parameters(),
                lr=1e-4,
                betas=(0.9, 0.999),
                weight_decay=1e-5
            )
            
            # Learning rate schedulers
            self.scheduler_encoder = optim.lr_scheduler.ReduceLROnPlateau(
                self.optimizer_encoder,
                mode='min',
                factor=0.5,
                patience=5
            )
            
            self.scheduler_decoder = optim.lr_scheduler.ReduceLROnPlateau(
                self.optimizer_decoder,
                mode='min',
                factor=0.5,
                patience=5
            )
            
            # Loss functions
            self.reconstruction_loss = nn.L1Loss()
            self.embedding_loss = nn.CosineEmbeddingLoss()
            
            # TensorBoard writer
            self.writer = SummaryWriter('runs/voice_cloning')
        
        def train_epoch(self, dataloader, epoch):
            """Train for one epoch"""
            self.encoder.train()
            self.decoder.train()
            
            total_loss = 0
            progress_bar = tqdm(dataloader, desc=f'Epoch {epoch}')
            
            for batch_idx, (mel_specs, _) in enumerate(progress_bar):
                mel_specs = mel_specs.to(self.device)
                
                # Zero gradients
                self.optimizer_encoder.zero_grad()
                self.optimizer_decoder.zero_grad()
                
                # Forward pass
                embeddings = self.encoder(mel_specs)
                reconstructed = self.decoder(embeddings, mel_specs.shape[2])
                
                # Calculate losses
                recon_loss = self.reconstruction_loss(reconstructed, mel_specs)
                
                # Embedding similarity loss (same speaker should have similar embeddings)
                embed_target = torch.ones(embeddings.shape[0]).to(self.device)
                embed_loss = self.embedding_loss(
                    embeddings[:-1],
                    embeddings[1:],
                    embed_target[:-1]
                )
                
                total_batch_loss = recon_loss + 0.1 * embed_loss
                
                # Backward pass
                total_batch_loss.backward()
                
                # Gradient clipping
                torch.nn.utils.clip_grad_norm_(self.encoder.parameters(), 5.0)
                torch.nn.utils.clip_grad_norm_(self.decoder.parameters(), 5.0)
                
                # Optimizer step
                self.optimizer_encoder.step()
                self.optimizer_decoder.step()
                
                total_loss += total_batch_loss.item()
                
                # Update progress bar
                progress_bar.set_postfix({
                    'loss': total_batch_loss.item(),
                    'recon': recon_loss.item(),
                    'embed': embed_loss.item()
                })
                
                # Log to TensorBoard
                global_step = epoch * len(dataloader) + batch_idx
                self.writer.add_scalar('Loss/train', total_batch_loss.item(), global_step)
                self.writer.add_scalar('Loss/reconstruction', recon_loss.item(), global_step)
                self.writer.add_scalar('Loss/embedding', embed_loss.item(), global_step)
            
            avg_loss = total_loss / len(dataloader)
            return avg_loss
        
        def train(self, dataloader, num_epochs=100):
            """Complete training loop"""
            best_loss = float('inf')
            
            for epoch in range(num_epochs):
                avg_loss = self.train_epoch(dataloader, epoch)
                
                print(f"\nEpoch {epoch} - Average Loss: {avg_loss:.4f}")
                
                # Update learning rate schedulers
                self.scheduler_encoder.step(avg_loss)
                self.scheduler_decoder.step(avg_loss)
                
                # Save checkpoint if best model
                if avg_loss < best_loss:
                    best_loss = avg_loss
                    self.save_checkpoint(epoch, avg_loss)
                    print(f"Saved best model with loss {best_loss:.4f}")
        
        def save_checkpoint(self, epoch, loss):
            """Save model checkpoint"""
            checkpoint = {
                'epoch': epoch,
                'encoder_state_dict': self.encoder.state_dict(),
                'decoder_state_dict': self.decoder.state_dict(),
                'optimizer_encoder_state_dict': self.optimizer_encoder.state_dict(),
                'optimizer_decoder_state_dict': self.optimizer_decoder.state_dict(),
                'loss': loss
            }
            torch.save(checkpoint, f'checkpoints/voice_cloning_epoch_{epoch}.pt')

    # Initialize trainer and train
    trainer = VoiceCloningTrainer(encoder, decoder)
    trainer.train(dataloader, num_epochs=100)
    ```

    ```python
    import torch
    import torchaudio
    from pathlib import Path

    class VoiceCloner:
        """Inference pipeline for voice cloning"""
        
        def __init__(self, encoder, decoder, processor, device='cuda'):
            self.encoder = encoder.to(device).eval()
            self.decoder = decoder.to(device).eval()
            self.processor = processor
            self.device = device
        
        @torch.no_grad()
        def extract_voice_embedding(self, reference_audio_path):
            """Extract voice embedding from reference audio"""
            # Process audio
            mel_spec = self.processor.process_audio(reference_audio_path)
            mel_spec = mel_spec.to(self.device)
            
            # Extract embedding
            embedding = self.encoder(mel_spec)
            
            return embedding
        
        @torch.no_grad()
        def clone_voice(self, reference_audio_path, target_length=256):
            """Clone voice from reference audio"""
            # Extract voice embedding
            embedding = self.extract_voice_embedding(reference_audio_path)
            
            # Generate new audio
            generated_mel = self.decoder(embedding, target_length)
            
            return generated_mel
        
        def mel_to_audio(self, mel_spec, output_path):
            """Convert mel spectrogram to audio waveform"""
            # Inverse mel transform (using Griffin-Lim or neural vocoder)
            vocoder = torchaudio.transforms.GriffinLim(
                n_fft=self.processor.n_fft,
                hop_length=self.processor.hop_length
            )
            
            # Denormalize and convert from log scale
            mel_spec = torch.exp(mel_spec) - 1e-9
            
            # Generate waveform
            waveform = vocoder(mel_spec)
            
            # Save audio
            torchaudio.save(
                output_path,
                waveform.cpu(),
                self.processor.sample_rate
            )
            
            return output_path
        
        def clone_and_save(self, reference_audio, output_path, target_length=256):
            """Complete cloning pipeline with audio output"""
            print(f"Cloning voice from {reference_audio}...")
            
            # Clone voice
            generated_mel = self.clone_voice(reference_audio, target_length)
            
            # Convert to audio and save
            output_file = self.mel_to_audio(generated_mel, output_path)
            
            print(f"Voice cloning complete! Saved to {output_file}")
            return output_file

    # Load trained models
    checkpoint = torch.load('checkpoints/voice_cloning_best.pt')
    encoder.load_state_dict(checkpoint['encoder_state_dict'])
    decoder.load_state_dict(checkpoint['decoder_state_dict'])

    # Initialize voice cloner
    cloner = VoiceCloner(encoder, decoder, processor)

    # Clone voice
    cloner.clone_and_save(
        reference_audio='reference_voice.wav',
        output_path='cloned_voice.wav',
        target_length=512
    )
    ```

## Model Evaluation & Testing

```python
import torch
from pesq import pesq
from pystoi import stoi
import numpy as np

class VoiceQualityEvaluator:
    """Evaluate synthesized voice quality"""
    
    def __init__(self, sample_rate=22050):
        self.sample_rate = sample_rate
    
    def calculate_pesq(self, reference, synthesized):
        """Calculate PESQ score (Perceptual Evaluation of Speech Quality)"""
        ref_np = reference.cpu().numpy().squeeze()
        synth_np = synthesized.cpu().numpy().squeeze()
        
        score = pesq(self.sample_rate, ref_np, synth_np, 'wb')
        return score
    
    def calculate_stoi(self, reference, synthesized):
        """Calculate STOI score (Short-Time Objective Intelligibility)"""
        ref_np = reference.cpu().numpy().squeeze()
        synth_np = synthesized.cpu().numpy().squeeze()
        
        score = stoi(ref_np, synth_np, self.sample_rate, extended=False)
        return score
    
    def evaluate_batch(self, reference_audios, synthesized_audios):
        """Evaluate multiple audio samples"""
        pesq_scores = []
        stoi_scores = []
        
        for ref, synth in zip(reference_audios, synthesized_audios):
            pesq_scores.append(self.calculate_pesq(ref, synth))
            stoi_scores.append(self.calculate_stoi(ref, synth))
        
        results = {
            'pesq_mean': np.mean(pesq_scores),
            'pesq_std': np.std(pesq_scores),
            'stoi_mean': np.mean(stoi_scores),
            'stoi_std': np.std(stoi_scores)
        }
        
        print(f"\nEvaluation Results:")
        print(f"PESQ: {results['pesq_mean']:.3f} ± {results['pesq_std']:.3f}")
        print(f"STOI: {results['stoi_mean']:.3f} ± {results['stoi_std']:.3f}")
        
        return results
```
